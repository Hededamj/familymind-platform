    ```tsx
    export async function generateMetadata(): Promise<Metadata> {
      const tenant = await getTenantConfig()
      return {
        title: tenant.brandName,
        description: tenant.description || tenant.tagline || '',
        manifest: '/manifest.json',
        other: {
          'apple-mobile-web-app-capable': 'yes',
          'apple-mobile-web-app-status-bar-style': 'default',
          'apple-mobile-web-app-title': tenant.brandName,
          'mobile-web-app-capable': 'yes',
        },
      }
    }
    ```

    Also add themeColor to the existing `viewport` export (currently on lines 39-43):

    ```tsx
    export const viewport: Viewport = {
      width: 'device-width',
      initialScale: 1,
      viewportFit: 'cover',
      themeColor: '#1A1A1A',
    }
    ```

    Notes:
    - `apple-mobile-web-app-capable: 'yes'` — iOS Safari "Add to Home Screen" will launch in standalone mode (no browser chrome). This satisfies PWA-02.
    - `apple-mobile-web-app-status-bar-style: 'default'` — shows the status bar in the default light style. Use 'black-translucent' only if the app header is dark and extends behind the status bar; 'default' is safer here.
    - `apple-mobile-web-app-title: tenant.brandName` — the name shown under the icon on iOS home screen.
    - `mobile-web-app-capable: 'yes'` — Android Chrome equivalent of apple-mobile-web-app-capable.
    - `themeColor: '#1A1A1A'` in viewport — Next.js renders this as `<meta name="theme-color" content="#1A1A1A">` which Android Chrome uses to color the browser chrome. This satisfies PWA-03.
    - The manifest field adds `<link rel="manifest" href="/manifest.json">` to the document head.
    - Do NOT change any other part of app/layout.tsx — only modify generateMetadata return and viewport export.
