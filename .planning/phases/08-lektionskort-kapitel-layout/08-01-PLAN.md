**Bookmark toggle logic (per CARD-03):**
```typescript
const [isSaved, setIsSaved] = useState(initialSaved)

async function handleBookmark(e: React.MouseEvent) {
  e.preventDefault()    // prevent Link navigation
  e.stopPropagation()   // prevent event bubbling to Link
  setIsSaved(!isSaved)  // optimistic toggle
  await toggleBookmarkAction(lesson.id, isSaved)
}
```
Import `toggleBookmarkAction` from `@/app/actions/savedContent`.
