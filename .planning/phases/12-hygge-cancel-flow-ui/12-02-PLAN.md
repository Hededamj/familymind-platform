**File 1: `app/dashboard/subscription/cancel/page.tsx`** (Server Component, no 'use client')

```tsx
import { redirect } from 'next/navigation'
import { requireAuth } from '@/lib/auth'
import { getUserEntitlements } from '@/lib/services/entitlement.service'
import { prisma } from '@/lib/prisma'
import { getStripe } from '@/lib/stripe'
import { CancelFlow } from './_components/cancel-flow'

export default async function CancelPage() {
  const user = await requireAuth()
  const entitlements = await getUserEntitlements(user.id)
  const active = entitlements.find((e) => e.status === 'ACTIVE' && e.stripeSubscriptionId)
  if (!active) redirect('/dashboard/settings')

  // Fetch period_end from Stripe for the confirm screen
  const stripe = getStripe()
  const sub = await stripe.subscriptions.retrieve(active.stripeSubscriptionId!)
  const periodEndUnix = (sub as unknown as { current_period_end: number }).current_period_end
  const periodEndDate = new Date(periodEndUnix * 1000)

  const reasons = await prisma.cancellationReason.findMany({
    orderBy: { slug: 'asc' },
    select: { slug: true, label: true },
  })

  const firstName = (user.name ?? '').trim().split(/\s+/)[0] || 'du'

  return (
    <div className="min-h-screen bg-[var(--color-sand)] px-4 py-8 sm:px-8">
      <div className="mx-auto w-full max-w-xl">
        <CancelFlow
          userName={firstName}
          entitlementId={active.id}
          periodEndDate={periodEndDate.toISOString()}
          priceAmountCents={active.priceVariant?.priceAmountCents ?? null}
          reasons={reasons}
        />
      </div>
    </div>
  )
}
```

```tsx
'use client'

import { useState, useTransition } from 'react'
import { useRouter } from 'next/navigation'
import type { EligibleOffer } from '@/lib/services/retention.service'
import { submitSurveyAndResolveOffer, confirmCancelAction, acceptOfferAction } from '../actions'
import { StepIntro } from './step-intro'
import { StepReason } from './step-reason'

export type FlowStep =
  | 'intro'
  | 'reason'
  | 'offer-discount'
  | 'offer-pause'
  | 'offer-support'
  | 'offer-content'
  | 'offer-skip'
  | 'confirm'
  | 'done-cancel'
  | 'done-pause'
  | 'done-retention'

export interface CancelFlowProps {
  userName: string
  entitlementId: string
  periodEndDate: string // ISO string — convert to Date in components that need formatting
  priceAmountCents: number | null
  reasons: { slug: string; label: string }[]
}

function routeToOfferStep(offerType: string | null): FlowStep {
  switch (offerType) {
    case 'DISCOUNT': return 'offer-discount'
    case 'PAUSE': return 'offer-pause'
    case 'SUPPORT': return 'offer-support'
    case 'CONTENT_HELP': return 'offer-content'
    case 'NONE': return 'offer-skip'
    default: return 'confirm'
  }
}

export function CancelFlow(props: CancelFlowProps) {
  const router = useRouter()
  const [isPending, startTransition] = useTransition()
  const [step, setStep] = useState<FlowStep>('intro')
  const [surveyId, setSurveyId] = useState<string | null>(null)
  const [offerId, setOfferId] = useState<string | null>(null)
  const [offerConfig, setOfferConfig] = useState<EligibleOffer | null>(null)
  const [acceptanceExpiresAt, setAcceptanceExpiresAt] = useState<Date | null>(null)
  const [error, setError] = useState<string | null>(null)

  const periodEndDate = new Date(props.periodEndDate)

  function handleSubmitReason(slugs: string[], feedback: string) {
    setError(null)
    startTransition(async () => {
      try {
        const res = await submitSurveyAndResolveOffer({
          entitlementId: props.entitlementId,
          reasonSlugs: slugs,
          feedback: feedback.trim() ? feedback.trim() : null,
          primaryReasonSlug: slugs[0],
        })
        setSurveyId(res.surveyId)
        setOfferId(res.offerId)
        setOfferConfig(res.offerConfig)
        setStep(routeToOfferStep(res.offerType))
      } catch (e) {
        setError(e instanceof Error ? e.message : 'Noget gik galt')
      }
    })
  }

  function handleConfirmCancel() {
    setError(null)
    startTransition(async () => {
      try {
        await confirmCancelAction({ entitlementId: props.entitlementId })
        setStep('done-cancel')
      } catch (e) {
        setError(e instanceof Error ? e.message : 'Noget gik galt')
      }
    })
  }

  function handleAcceptOffer(pauseMonths?: 1 | 2 | 3) {
    if (!surveyId || !offerId) return
    setError(null)
    startTransition(async () => {
      try {
        const res = await acceptOfferAction({
          surveyId,
          offerId,
          entitlementId: props.entitlementId,
          pauseMonths,
        })
        setAcceptanceExpiresAt(res.expiresAt)
        setStep(pauseMonths ? 'done-pause' : 'done-retention')
      } catch (e) {
        setError(e instanceof Error ? e.message : 'Noget gik galt')
      }
    })
  }

  function handleReturnToDashboard() {
    router.push('/dashboard')
  }

  return (
    <>
      {error && (
        <div className="mb-4 rounded-xl border border-destructive/50 bg-destructive/10 p-4 text-sm text-destructive">
          {error}
        </div>
      )}

      {step === 'intro' && (
        <StepIntro
          onContinue={() => setStep('reason')}
          onSkip={() => setStep('confirm')}
        />
      )}

      {step === 'reason' && (
        <StepReason
          reasons={props.reasons}
          loading={isPending}
          onBack={() => setStep('intro')}
          onSubmit={handleSubmitReason}
        />
      )}

      {/* Plan 12-03 replaces these placeholders */}
      {(step === 'offer-discount' ||
        step === 'offer-pause' ||
        step === 'offer-support' ||
        step === 'offer-content' ||
        step === 'offer-skip' ||
        step === 'confirm' ||
        step === 'done-cancel' ||
        step === 'done-pause' ||
        step === 'done-retention') && (
        <div className="rounded-2xl border border-border bg-white p-8 text-center">
          <p className="text-sm text-muted-foreground">TODO Plan 12-03: {step}</p>
          <button
            onClick={handleReturnToDashboard}
            className="mt-4 text-sm underline"
          >
            Tilbage til forsiden
          </button>
        </div>
      )}

      {/* Keep handler references used-by-typescript so Plan 12-03 wiring compiles */}
      <span className="hidden" aria-hidden>
        {typeof handleConfirmCancel}{typeof handleAcceptOffer}{String(acceptanceExpiresAt)}{String(offerConfig)}{String(periodEndDate)}{props.userName}{String(props.priceAmountCents)}
      </span>
    </>
  )
}
```

```tsx
'use client'

import { useState } from 'react'

export interface StepReasonProps {
  reasons: { slug: string; label: string }[]
  loading: boolean
  onBack: () => void
  onSubmit: (slugs: string[], feedback: string) => void
}

// Canonical ordering + labels per research doc. We display this order regardless of DB order.
const DISPLAY_ORDER: { slug: string; label: string }[] = [
  { slug: 'pris', label: 'Prisen' },
  { slug: 'tid', label: 'For lidt tid' },
  { slug: 'indhold-matcher-ikke', label: 'Matcher ikke mig' },
  { slug: 'personlig-situation', label: 'Personlig situation' },
  { slug: 'fandt-alternativ', label: 'Fandt noget andet' },
  { slug: 'forbedret', label: 'Jeg har det bedre nu' },
  { slug: 'teknisk', label: 'Tekniske problemer' },
]

export function StepReason({ reasons, loading, onBack, onSubmit }: StepReasonProps) {
  const [selected, setSelected] = useState<Set<string>>(new Set())
  const [feedback, setFeedback] = useState('')

  // Merge DB reasons with display order — only show slugs that exist in DB, keep display labels
  const available = DISPLAY_ORDER.filter((d) => reasons.some((r) => r.slug === d.slug))

  function toggle(slug: string) {
    setSelected((prev) => {
      const next = new Set(prev)
      if (next.has(slug)) next.delete(slug)
      else next.add(slug)
      return next
    })
  }

  const canSubmit = selected.size > 0 && !loading

  return (
    <div className="rounded-2xl border border-border bg-white p-6 sm:p-8">
      <button
        type="button"
        onClick={onBack}
        className="mb-4 inline-flex items-center gap-1 text-sm text-muted-foreground hover:text-foreground"
      >
        ← Tilbage
      </button>
      <h2 className="mb-2 font-serif text-2xl sm:text-3xl">
        Hvad fik dig til at overveje at stoppe?
      </h2>
      <p className="mb-6 text-sm text-muted-foreground">
        Vælg hvad der passer — du må vælge flere
      </p>

      <div className="mb-6 flex flex-wrap gap-2">
        {available.map((r) => {
          const isSelected = selected.has(r.slug)
          return (
            <button
              key={r.slug}
              type="button"
              onClick={() => toggle(r.slug)}
              className={
                isSelected
                  ? 'rounded-full border border-[var(--color-coral)] bg-[var(--color-coral)] px-4 py-2.5 text-sm font-medium text-white transition-colors'
                  : 'rounded-full border border-border bg-white px-4 py-2.5 text-sm font-medium transition-colors hover:border-[var(--color-coral)]'
              }
              aria-pressed={isSelected}
            >
              {r.label}
            </button>
          )
        })}
      </div>

      <label className="mb-2 block text-sm font-medium">
        Noget andet du gerne vil fortælle os?
      </label>
      <p className="mb-2 text-xs text-muted-foreground">
        Helt valgfrit — men vi læser alt
      </p>
      <textarea
        value={feedback}
        onChange={(e) => setFeedback(e.target.value)}
        rows={4}
        className="mb-6 w-full resize-none rounded-xl border border-border bg-white p-3 text-sm outline-none focus:border-[var(--color-coral)]"
      />

      <button
        type="button"
        disabled={!canSubmit}
        onClick={() => onSubmit(Array.from(selected), feedback)}
        className="w-full rounded-2xl bg-[var(--color-coral)] px-6 py-4 text-sm font-semibold text-white hover:opacity-90 disabled:cursor-not-allowed disabled:opacity-50"
      >
        {loading ? 'Gemmer…' : 'Næste'}
      </button>
    </div>
  )
}
```
